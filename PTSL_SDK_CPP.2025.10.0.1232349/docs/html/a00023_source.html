132 std::tuple<bool, std::string> ConvertJsonToResponseError(const std::string& errorJSON)
133 {
134 JsonParseOptions jOpts = DefaultJsonParseOptions();
135
136 auto status = JsonStringToMessage(errorJSON, &mGrpcResponseError, jOpts);
137 bool result = status.ok();
138 std::string result_str = status.ToString();
139
140 if (!status.ok())
141 {
142 ptsl::CommandError legacyError;
143 auto legacy_err_status = JsonStringToMessage(errorJSON, &legacyError);
144 result = legacy_err_status.ok();
145 if (legacy_err_status.ok())
146 {
147 mGrpcResponseError.add_errors()->CopyFrom(legacyError);
148 // Reset result_str to the legacy error status since
149 // the previous error was expected
150 result_str = legacy_err_status.ToString();
151 }
152 else
153 {
154 // Append the legacy error status to the result string
155 // for additional context
156 result_str += ";" + legacy_err_status.ToString();
157 }
158 }
159
160 return { result, result_str };
161 }
