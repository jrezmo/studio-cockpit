54 virtual std::string ToString() const
55 {
56 if (!gotResponse)
57 {
58 return "";
59 }
60
61 auto isNullOrWhiteSpace = [](const std::string& str)
62 { return str.empty() || std::all_of(str.begin(), str.end(), ::isspace); };
63
64 json responseJson;
65 auto emptyJson = json::object();
66
67 responseJson[Constants::RESPONSE_HEADER] = isNullOrWhiteSpace(jsonHeader) ? emptyJson : json::parse(jsonHeader);
68 responseJson[Constants::RESPONSE_BODY] = isNullOrWhiteSpace(jsonBody) ? emptyJson : json::parse(jsonBody);
69 responseJson[Constants::RESPONSE_ERROR] = isNullOrWhiteSpace(jsonError) ? emptyJson : json::parse(jsonError);
70
71 return responseJson.dump(indent) + "\n";
72 }
